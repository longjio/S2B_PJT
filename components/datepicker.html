DatePicker (Tempus Dominus)

Tempus Dominus v6를 사용한 날짜/시간 선택 컴포넌트 - Bootstrap 5와 완벽 호환 (jQuery 불필요!)

기본 날짜 선택기

<!-- HTML -->
<div class="input-group" id="basicDatepicker" data-td-target-input="nearest" data-td-target-toggle="nearest">
  <input type="text" class="form-control" data-td-target="#basicDatepicker" placeholder="날짜를 선택하세요"/>
  <span class="input-group-text" data-td-target="#basicDatepicker" data-td-toggle="datetimepicker">
    <i class="bi bi-calendar3"></i>
  </span>
</div>

<!-- JavaScript -->
<script type="module">
import { TempusDominus } from '@eonasdan/tempus-dominus';

new TempusDominus(document.getElementById('basicDatepicker'), {
  localization: {
    locale: 'ko',
    format: 'yyyy-MM-dd'
  }
});
</script>

날짜 범위 선택기

<!-- 시작 날짜 -->
<div class="input-group" id="startDate" data-td-target-input="nearest" data-td-target-toggle="nearest">
  <input type="text" class="form-control" data-td-target="#startDate" placeholder="시작 날짜"/>
  <span class="input-group-text" data-td-target="#startDate" data-td-toggle="datetimepicker">
    <i class="bi bi-calendar-range"></i>
  </span>
</div>

<!-- 종료 날짜 -->
<div class="input-group" id="endDate" data-td-target-input="nearest" data-td-target-toggle="nearest">
  <input type="text" class="form-control" data-td-target="#endDate" placeholder="종료 날짜"/>
  <span class="input-group-text" data-td-target="#endDate" data-td-toggle="datetimepicker">
    <i class="bi bi-calendar-range"></i>
  </span>
</div>

<script type="module">
import { TempusDominus } from '@eonasdan/tempus-dominus';

const startPicker = new TempusDominus(document.getElementById('startDate'), {
  localization: {
    locale: 'ko',
    format: 'yyyy-MM-dd'
  }
});

const endPicker = new TempusDominus(document.getElementById('endDate'), {
  localization: {
    locale: 'ko',
    format: 'yyyy-MM-dd'
  },
  useCurrent: false
});

// 시작일이 변경되면 종료일의 최소값 설정
startPicker.subscribe('change.td', (e) => {
  endPicker.updateOptions({
    restrictions: {
      minDate: e.date
    }
  });
});

// 종료일이 변경되면 시작일의 최대값 설정
endPicker.subscribe('change.td', (e) => {
  startPicker.updateOptions({
    restrictions: {
      maxDate: e.date
    }
  });
});
</script>

통합 날짜 범위 선택기

<!-- HTML -->
<div class="input-group">
  <input type="text" id="combinedDateRangeInput" class="form-control" placeholder="시작일 ~ 종료일" readonly/>
  <button class="input-group-text" id="combinedDateRangeBtn" type="button">
    <i class="bi bi-calendar-range"></i>
  </button>
</div>
<div id="combinedDateRangeContainer" class="date-range-picker-container" style="display: none; margin-top: 10px;">
  <div class="row g-2">
    <div class="col-6">
      <div id="combinedStartDate"></div>
    </div>
    <div class="col-6">
      <div id="combinedEndDate"></div>
    </div>
  </div>
</div>

<!-- JavaScript -->
<script type="module">
import { TempusDominus } from '@eonasdan/tempus-dominus';

const input = document.getElementById('combinedDateRangeInput');
const btn = document.getElementById('combinedDateRangeBtn');
const container = document.getElementById('combinedDateRangeContainer');

const startPicker = new TempusDominus(document.getElementById('combinedStartDate'), {
  display: { inline: true }
});

const endPicker = new TempusDominus(document.getElementById('combinedEndDate'), {
  display: { inline: true }
});

// 토글 기능
btn.addEventListener('click', () => {
  container.style.display = container.style.display === 'none' ? 'block' : 'none';
});

// 날짜 선택 시 input 업데이트
function updateInput() {
  const start = startPicker.dates.lastPicked;
  const end = endPicker.dates.lastPicked;
  if (start && end) {
    input.value = start.format('YYYY-MM-DD') + ' ~ ' + end.format('YYYY-MM-DD');
  }
}

startPicker.subscribe('change.td', updateInput);
endPicker.subscribe('change.td', updateInput);
</script>

특정 날짜 비활성화

주말은 선택할 수 없습니다 
<div class="input-group" id="restrictedDatepicker" data-td-target-input="nearest" data-td-target-toggle="nearest">
  <input type="text" class="form-control" data-td-target="#restrictedDatepicker" placeholder="평일만 선택 가능"/>
  <span class="input-group-text" data-td-target="#restrictedDatepicker" data-td-toggle="datetimepicker">
    <i class="bi bi-calendar-x"></i>
  </span>
</div>

<script type="module">
import { TempusDominus } from '@eonasdan/tempus-dominus';

new TempusDominus(document.getElementById('restrictedDatepicker'), {
  restrictions: {
    daysOfWeekDisabled: [0, 6] // 일요일(0), 토요일(6) 비활성화
  },
  localization: {
    locale: 'ko',
    format: 'yyyy-MM-dd'
  }
});
</script>
