Tables (Tabulator)

Tabulator 라이브러리를 사용한 고급 테이블 - 검색, 정렬, 페이지네이션 기능 제공 (Vanilla JS, jQuery 불필요!)

게시판 스타일 테이블

<!-- HTML -->
<div id="boardTable"></div>

<!-- JavaScript -->
<script>
const boardData = [
  { id: 15, title: 'Bootstrap Table 사용법', author: '관리자', date: '2025-01-15', views: 245 },
  { id: 14, title: '공지사항입니다', author: '운영자', date: '2025-01-14', views: 189 },
  // ... more data
];

new Tabulator("#boardTable", {
  data: boardData,
  layout: "fitColumns",
  pagination: true,
  paginationSize: 10,
  paginationSizeSelector: [10, 20, 50],
  columns: [
    { title: "번호", field: "id", sorter: "number", width: 80 },
    { title: "제목", field: "title", sorter: "string" },
    { title: "작성자", field: "author", sorter: "string", width: 120 },
    { title: "작성일", field: "date", sorter: "date", width: 120 },
    { title: "조회", field: "views", sorter: "number", width: 80 }
  ]
});
</script>

컴팩트 테이블

<div id="compactTable"></div>

<script>
const compactData = [
  { no: 1, name: '홍길동', email: 'hong@example.com', status: '활성' },
  { no: 2, name: '김철수', email: 'kim@example.com', status: '활성' },
  // ... more data
];

new Tabulator("#compactTable", {
  data: compactData,
  layout: "fitColumns",
  pagination: true,
  paginationSize: 5,
  columns: [
    { title: "No", field: "no", sorter: "number", width: 80 },
    { title: "이름", field: "name", sorter: "string", width: 120 },
    { title: "이메일", field: "email", sorter: "string" },
    { title: "상태", field: "status", sorter: "string", width: 100 }
  ]
});
</script>
